89/*
90 * The C, U, and N flags indicate the properties
91 * Critical, Unsafe, and NoCacheKey, respectively.
92 * If U is set, then N has no meaning as per
93 * https://rfc-editor.org/rfc/rfc7252#section-5.10
94 * and is set to a -.
95 *
96 * Separately, R is for the options that can be repeated
97 *
98 * The least significant byte of the option is set as followed
99 * as per https://rfc-editor.org/rfc/rfc7252#section-5.4.6
100 *
101 * 0 1 2 3 4 5 6 7
102 * --+---+---+---+---+---+---+---+
103 * | NoCacheKey| U | C |
104 * --+---+---+---+---+---+---+---+
105 *
106 * https://rfc-editor.org/rfc/rfc8613#section-4 goes on to define E, I and U
107 * properties Encrypted and Integrity Protected, Integrity Protected Only, and
108 * Unprotected respectively. Integrity Protected Only is not currently used.
109 *
110 * An Option is tagged with CUNREIU with any of the letters replaced with _ if
111 * not set, or - for N if U is set (see above) for aiding understanding of the
112 * Option.
113 */
114

146/* CoAP result codes (HTTP-Code / 100 * 40 + HTTP-Code % 100) */
147
148/* As of draft-ietf-core-coap-04, response codes are encoded to base
149 * 32, i.e. the three upper bits determine the response class while
150 * the remaining five fine-grained information specific to that class.
151 */
152#define COAP_RESPONSE_CODE(N) (((N)/100 << 5) | (N)%100)
153
154/* Determines the class of response code C */
155#define COAP_RESPONSE_CLASS(C) (((C) >> 5) & 0xFF)
