31 u32_t v = LWIP_RAND();
32 while (len > sizeof(v)) {
33 memcpy(buf, &v, sizeof(v));
34 len -= sizeof(v);
35 buf += sizeof(v);
36 v = LWIP_RAND();
37 }
38
39 memcpy(buf, &v, len);
40 return 1;
