85/*
86 * The C, U, and N flags indicate the properties
87 * Critical, Unsafe, and NoCacheKey, respectively.
88 * If U is set, then N has no meaning as per
89 * https://tools.ietf.org/html/rfc7252#section-5.10
90 * and is set to a -.
91 *
92 * Separately, R is for the options that can be repeated
93 *
94 * The least significant byte of the option is set as followed
95 * as per https://tools.ietf.org/html/rfc7252#section-5.4.6
96 *
97 * 0 1 2 3 4 5 6 7
98 * --+---+---+---+---+---+---+---+
99 * | NoCacheKey| U | C |
100 * --+---+---+---+---+---+---+---+
101 *
102 * https://tools.ietf.org/html/rfc8613#section-4 goes on to define E, I and U
103 * properties Encrypted and Integrity Protected, Integrity Protected Only, and
104 * Unprotected respectively. Integrity Protected Only is not currently used.
105 *
106 * An Option is tagged with CUNREIU with any of the letters replaced with _ if
107 * not set, or - for N if U is set (see above) for aiding understanding of the
108 * Option.
109 */
110

142/* As of draft-ietf-core-coap-04, response codes are encoded to base
143 * 32, i.e. the three upper bits determine the response class while
144 * the remaining five fine-grained information specific to that class.
145 */
146#define COAP_RESPONSE_CODE(N) (((N)/100 << 5) | (N)%100)
147
148/* Determines the class of response code C */
149#define COAP_RESPONSE_CLASS(C) (((C) >> 5) & 0xFF)
