35COAP_STATIC_INLINE int
36contiki_prng_impl(unsigned char *buf, size_t len) {
37 uint16_t v = random_rand();
38 while (len > sizeof(v)) {
39 memcpy(buf, &v, sizeof(v));
40 len -= sizeof(v);
41 buf += sizeof(v);
42 v = random_rand();
43 }
44
45 memcpy(buf, &v, len);
46 return 1;
