44unsigned int
45coap_encode_var_safe(uint8_t *buf, size_t length, unsigned int val) {
46 unsigned int n, i;
47
48 for (n = 0, i = val; i && n < sizeof(val); ++n)
49 i >>= 8;
50
51 if (n > length) {
52 assert (n <= length);
53 return 0;
54 }
55 i = n;
56 while (i--) {
57 buf[i] = val & 0xff;
58 val >>= 8;
59 }
60
61 return n;
62}
