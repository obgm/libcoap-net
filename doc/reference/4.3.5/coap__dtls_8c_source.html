25void
26coap_dtls_map_key_type_to_define(const coap_dtls_pki_t *setup_data, coap_dtls_key_t *key) {
27 *key = setup_data->pki_key;
28
29 switch (key->key_type) {
30 case COAP_PKI_KEY_PEM:
31 key->key_type = COAP_PKI_KEY_DEFINE;
32 key->key.define.ca.s_byte = setup_data->pki_key.key.pem.ca_file;
33 key->key.define.public_cert.s_byte = setup_data->pki_key.key.pem.public_cert;
34 key->key.define.private_key.s_byte = setup_data->pki_key.key.pem.private_key;
35
36 key->key.define.public_cert_def = COAP_PKI_KEY_DEF_PEM;
37 key->key.define.private_key_def = COAP_PKI_KEY_DEF_PEM;
38 key->key.define.ca_def = COAP_PKI_KEY_DEF_PEM;
39 break;
40 case COAP_PKI_KEY_ASN1:
41 key->key_type = COAP_PKI_KEY_DEFINE;
42 key->key.define.ca.u_byte = setup_data->pki_key.key.asn1.ca_cert;
43 key->key.define.public_cert.u_byte = setup_data->pki_key.key.asn1.public_cert;
44 key->key.define.private_key.u_byte = setup_data->pki_key.key.asn1.private_key;
45
46 key->key.define.ca_len = setup_data->pki_key.key.asn1.ca_cert_len;
47 key->key.define.public_cert_len = setup_data->pki_key.key.asn1.public_cert_len;
48 key->key.define.private_key_len = setup_data->pki_key.key.asn1.private_key_len;
49
50 key->key.define.private_key_type = setup_data->pki_key.key.asn1.private_key_type;
51
52 key->key.define.public_cert_def = COAP_PKI_KEY_DEF_DER_BUF;
53 key->key.define.private_key_def = COAP_PKI_KEY_DEF_DER_BUF;
54 key->key.define.ca_def = COAP_PKI_KEY_DEF_DER_BUF;
55 break;
56 case COAP_PKI_KEY_PEM_BUF:
57 key->key_type = COAP_PKI_KEY_DEFINE;
58 key->key.define.ca.u_byte = setup_data->pki_key.key.pem_buf.ca_cert;
59 key->key.define.public_cert.u_byte = setup_data->pki_key.key.pem_buf.public_cert;
60 key->key.define.private_key.u_byte = setup_data->pki_key.key.pem_buf.private_key;
61
62 key->key.define.ca_len = setup_data->pki_key.key.pem_buf.ca_cert_len;
63 key->key.define.public_cert_len = setup_data->pki_key.key.pem_buf.public_cert_len;
64 key->key.define.private_key_len = setup_data->pki_key.key.pem_buf.private_key_len;
65
66 if (setup_data->is_rpk_not_cert) {
67 key->key.define.public_cert_def = COAP_PKI_KEY_DEF_RPK_BUF;
68 } else {
69 key->key.define.public_cert_def = COAP_PKI_KEY_DEF_PEM_BUF;
70 }
71 if (setup_data->is_rpk_not_cert) {
72 key->key.define.private_key_def = COAP_PKI_KEY_DEF_RPK_BUF;
73 } else {
74 key->key.define.private_key_def = COAP_PKI_KEY_DEF_PEM_BUF;
75 }
76 if (setup_data->is_rpk_not_cert) {
77 key->key.define.ca_def = COAP_PKI_KEY_DEF_RPK_BUF;
78 } else {
79 key->key.define.ca_def = COAP_PKI_KEY_DEF_PEM_BUF;
80 }
81 break;
82 case COAP_PKI_KEY_PKCS11:
83 key->key_type = COAP_PKI_KEY_DEFINE;
84 key->key.define.ca.s_byte = setup_data->pki_key.key.pkcs11.ca;
85 key->key.define.public_cert.s_byte = setup_data->pki_key.key.pkcs11.public_cert;
86 key->key.define.private_key.s_byte = setup_data->pki_key.key.pkcs11.private_key;
87
88 key->key.define.user_pin = setup_data->pki_key.key.pkcs11.user_pin;
89
90 if (strncasecmp(key->key.pkcs11.ca, "pkcs11:", 7) == 0) {
91 if (setup_data->is_rpk_not_cert) {
92 key->key.define.ca_def = COAP_PKI_KEY_DEF_PKCS11_RPK;
93 } else {
94 key->key.define.ca_def = COAP_PKI_KEY_DEF_PKCS11;
95 }
96 } else {
97 if (setup_data->is_rpk_not_cert) {
98 key->key.define.ca_def = COAP_PKI_KEY_DEF_RPK_BUF;
99 } else {
100 key->key.define.ca_def = COAP_PKI_KEY_DEF_DER;
101 }
102 }
103 if (strncasecmp(key->key.pkcs11.public_cert, "pkcs11:", 7) == 0) {
104 if (setup_data->is_rpk_not_cert) {
105 key->key.define.public_cert_def = COAP_PKI_KEY_DEF_PKCS11_RPK;
106 } else {
107 key->key.define.public_cert_def = COAP_PKI_KEY_DEF_PKCS11;
108 }
109 } else {
110 if (setup_data->is_rpk_not_cert) {
111 key->key.define.public_cert_def = COAP_PKI_KEY_DEF_RPK_BUF;
112 } else {
113 key->key.define.public_cert_def = COAP_PKI_KEY_DEF_DER;
114 }
115 }
116 if (strncasecmp(key->key.pkcs11.private_key, "pkcs11:", 7) == 0) {
117 if (setup_data->is_rpk_not_cert) {
118 key->key.define.private_key_def = COAP_PKI_KEY_DEF_PKCS11_RPK;
119 } else {
120 key->key.define.private_key_def = COAP_PKI_KEY_DEF_PKCS11;
121 }
122 } else {
123 if (setup_data->is_rpk_not_cert) {
124 key->key.define.private_key_def = COAP_PKI_KEY_DEF_RPK_BUF;
125 } else {
126 key->key.define.private_key_def = COAP_PKI_KEY_DEF_DER;
127 }
128 }
129 break;
130 case COAP_PKI_KEY_DEFINE:
131 /* Already configured */
132 break;
133 default:
134 break;
135 }
136}

164int
165coap_dtls_define_issue(coap_define_issue_key_t type, coap_define_issue_fail_t fail,
166 coap_dtls_key_t *key, const coap_dtls_role_t role, int ret) {
167#if (COAP_MAX_LOGGING_LEVEL >= _COAP_LOG_ERR)
168 coap_pki_key_define_t define = key->key.define;
169 switch (type) {
170 case COAP_DEFINE_KEY_CA:
171 switch (fail) {
172 case COAP_DEFINE_FAIL_BAD:
173 coap_log_warn("*** setup_pki: (D)TLS: %s: %s CA configure failure\n",
174 coap_dtls_get_define_type(define.ca_def, define.ca),
175 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
176 break;
177 case COAP_DEFINE_FAIL_NOT_SUPPORTED:
178 coap_log_err("*** setup_pki: (D)TLS: %s: %s CA type not supported\n",
179 coap_dtls_get_define_type(define.ca_def, define.ca),
180 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
181 break;
182 case COAP_DEFINE_FAIL_NONE:
183 coap_log_err("*** setup_pki: (D)TLS: %s: %s CA not defined\n",
184 coap_dtls_get_define_type(define.ca_def, define.ca),
185 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
186 break;
187 default:
188 break;
189 }
190 break;
191 case COAP_DEFINE_KEY_ROOT_CA:
192 switch (fail) {
193 case COAP_DEFINE_FAIL_BAD:
194 coap_log_warn("*** setup_pki: (D)TLS: %s: %s Root CA configure failure\n",
195 coap_dtls_get_define_type(define.ca_def, define.ca),
196 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
197 break;
198 case COAP_DEFINE_FAIL_NOT_SUPPORTED:
199 coap_log_err("*** setup_pki: (D)TLS: %s: %s Root CA type not supported\n",
200 coap_dtls_get_define_type(define.ca_def, define.ca),
201 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
202 break;
203 case COAP_DEFINE_FAIL_NONE:
204 coap_log_err("*** setup_pki: (D)TLS: %s: %s Root CA not defined\n",
205 coap_dtls_get_define_type(define.ca_def, define.ca),
206 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
207 break;
208 default:
209 break;
210 }
211 break;
212 case COAP_DEFINE_KEY_PUBLIC:
213 switch (fail) {
214 case COAP_DEFINE_FAIL_BAD:
215 coap_log_warn("*** setup_pki: (D)TLS: %s: %s Certificate configure failure\n",
216 coap_dtls_get_define_type(define.public_cert_def, define.public_cert),
217 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
218 break;
219 case COAP_DEFINE_FAIL_NOT_SUPPORTED:
220 coap_log_err("*** setup_pki: (D)TLS: %s: %s Certificate type not supported\n",
221 coap_dtls_get_define_type(define.public_cert_def, define.public_cert),
222 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
223 break;
224 case COAP_DEFINE_FAIL_NONE:
225 coap_log_err("*** setup_pki: (D)TLS: %s: %s Certificate not defined\n",
226 coap_dtls_get_define_type(define.public_cert_def, define.public_cert),
227 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
228 break;
229 default:
230 break;
231 }
232 break;
233 case COAP_DEFINE_KEY_PRIVATE:
234 switch (fail) {
235 case COAP_DEFINE_FAIL_BAD:
236 coap_log_warn("*** setup_pki: (D)TLS: %s: %s Private Key configure failure\n",
237 coap_dtls_get_define_type(define.private_key_def, define.private_key),
238 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
239 break;
240 case COAP_DEFINE_FAIL_NOT_SUPPORTED:
241 coap_log_err("*** setup_pki: (D)TLS: %s: %s Private Key type not supported\n",
242 coap_dtls_get_define_type(define.private_key_def, define.private_key),
243 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
244 break;
245 case COAP_DEFINE_FAIL_NONE:
246 coap_log_err("*** setup_pki: (D)TLS: %s: %s Private Key not defined\n",
247 coap_dtls_get_define_type(define.private_key_def, define.private_key),
248 role == COAP_DTLS_ROLE_SERVER ? "Server" : "Client");
249 break;
250 default:
251 break;
252 }
253 default:
254 break;
255 }
256#else /* COAP_MAX_LOGGING_LEVEL >= _COAP_LOG_ERR */
257 (void)type;
258 (void)fail;
259 (void)key;
260 (void)role;
261#endif /* COAP_MAX_LOGGING_LEVEL < _COAP_LOG_ERR */
262 return ret;

265void
266coap_dtls_establish(coap_session_t *session) {
267 session->state = COAP_SESSION_STATE_HANDSHAKE;
268#if COAP_CLIENT_SUPPORT
269 if (session->type == COAP_SESSION_TYPE_CLIENT)
270 session->tls = coap_dtls_new_client_session(session);
271#endif /* COAP_CLIENT_SUPPORT */
272#if COAP_SERVER_SUPPORT
273 if (session->type != COAP_SESSION_TYPE_CLIENT)
274 session->tls = coap_dtls_new_server_session(session);
275#endif /* COAP_SERVER_SUPPORT */
276
277 if (!session->tls) {
278 coap_session_disconnected_lkd(session, COAP_NACK_TLS_LAYER_FAILED);
279 return;
280 }
281 coap_ticks(&session->last_rx_tx);
282}

284void
285coap_dtls_close(coap_session_t *session) {

287 coap_dtls_free_session(session);
288 session->tls = NULL;
289 }
290 session->sock.lfunc[COAP_LAYER_TLS].l_close(session);
