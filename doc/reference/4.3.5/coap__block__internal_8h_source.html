65#define set_block_mode_probe_q(block_mode) \

67 block_mode |= COAP_BLOCK_PROBE_Q_BLOCK; \
68 block_mode &= ~(COAP_BLOCK_TRY_Q_BLOCK | COAP_BLOCK_HAS_Q_BLOCK); \
